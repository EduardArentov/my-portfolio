// ========== 1. КНОПКА "ЧИТАТЬ ДАЛЕЕ" ==========
// Показывает/скрывает дополнительный текст

const readMoreBtn = document.getElementById('readMoreBtn');
const extraText = document.getElementById('extraText');

let isExpanded = false; // переменная для отслеживания состояния

readMoreBtn.addEventListener('click', function() {
    if (isExpanded) {
        extraText.style.display = 'none';
        readMoreBtn.textContent = '📖 Читать далее';
        isExpanded = false;
    } else {
        extraText.style.display = 'block';
        readMoreBtn.textContent = '📖 Свернуть';
        isExpanded = true;
    }
});

// ========== 2. СЧЁТЧИК ЛАЙКОВ (сохраняется в localStorage) ==========

const likeBtn = document.getElementById('likeBtn');
const likeCountSpan = document.getElementById('likeCount');

// Загружаем сохранённое количество лайков (если есть)
let likeCount = localStorage.getItem('portfolioLikes');
if (likeCount === null) {
    likeCount = 0;
} else {
    likeCount = Number(likeCount);
}
likeCountSpan.textContent = likeCount;

// Обработчик нажатия на лайк
likeBtn.addEventListener('click', function() {
    likeCount++;
    likeCountSpan.textContent = likeCount;
    localStorage.setItem('portfolioLikes', likeCount);
    
    // Анимация кнопки
    likeBtn.style.transform = 'scale(1.2)';
    setTimeout(() => {
        likeBtn.style.transform = 'scale(1)';
    }, 200);
});

// ========== 4. КЛИК ПО НАВЫКАМ (выводит сообщение) ==========

const skillItems = document.querySelectorAll('.skills-list li');

skillItems.forEach(function(skill) {
    skill.addEventListener('click', function() {
        const skillName = this.textContent;
        alert(`🔧 Ты кликнул по навыку: ${skillName}! Молодец, продолжай развиваться!`);
    });
});

// ========== 5. КЛИК ПО КАРТОЧКАМ ПРОЕКТОВ ==========

const projectCards = document.querySelectorAll('.project-card');

});
